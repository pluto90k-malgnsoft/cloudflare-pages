packages = ["m3u8", "ssl", "certifi"] import asyncio import m3u8 import urllib.parse from js import window, document, Uint8Array from pyodide.http import pyfetch def log(msg, type=None): window.addLog(msg, type) async def start_analysis(event): btn = document.getElementById("btn-run") btn.disabled = True document.getElementById("loader").style.display = "inline-block" document.getElementById("status-text").innerText = "분석 시작 중..." document.getElementById("progress-container").style.display = "block" document.getElementById("output-container").innerHTML = "" window.updateProgress(0) url = document.getElementById("m3u8-url").value if not url: log("HLS URL을 입력해 주세요.", "error") reset_ui() return max_segs = int(document.getElementById("max-segs").value) proxy_prefix = document.getElementById("proxy-prefix").value.strip() try: await check_hls_continuity(url, max_segs, proxy_prefix) except Exception as e: log(f"분석 중 치명적 오류: {str(e)}", "error") finally: reset_ui() def reset_ui(): document.getElementById("btn-run").disabled = False document.getElementById("loader").style.display = "none" document.getElementById("status-text").innerText = "완료" def wrap_proxy(url, prefix): if not prefix: return url # 만약 prefix 끝에 url 파라미터가 명시되어 있다면 인코딩해서 붙임 (예: ?url=) if prefix.endswith("="): return prefix + urllib.parse.quote(url) # 그 외에는 그냥 단순 문자열 합치기 (예: https://proxy.com/) return prefix + url async def fetch_with_proxy(url, prefix): final_url = wrap_proxy(url, prefix) try: res = await pyfetch(final_url) if res.ok: return res raise Exception(f"HTTP {res.status}") except Exception as e: raise Exception(f"네트워크/CORS 오류: {final_url[:60]}... ({e})") async def check_hls_continuity(m3u8_url, max_segments, proxy_prefix): log(f"[*] 분석 시작: {m3u8_url}", "header") try: resp = await fetch_with_proxy(m3u8_url, proxy_prefix) content = await resp.string() playlist = m3u8.loads(content, uri=m3u8_url) except Exception as e: log(f"플레이리스트 로드 실패: {e}", "error") return if playlist.is_variant: log("[*] 마스터 플레이리스트 감지. 스트림 선택 중...", "info") media_url = playlist.playlists[0].absolute_uri resp = await fetch_with_proxy(media_url, proxy_prefix) content = await resp.string() playlist = m3u8.loads(content, uri=media_url) segments = playlist.segments if not segments: log("세그먼트 정보를 찾을 수 없습니다.", "error") return count = min(len(segments), max_segments) log(f"[*] 총 {len(segments)}개 중 {count}개 분석", "info") # 암호화 확인 if any(seg.key for seg in segments): log("[!] 암호화된(AES-128 등) 세그먼트가 감지되었습니다. 현재 분석기는 암호화 해제를 지원하지 않아 결과가 부정확할 수 있습니다.", "warning") last_dts_end = None for i in range(count): document.getElementById("status-text").innerText = f"분석 중 ({i+1}/{count})" seg = segments[i] log(f"Seg {i}: {seg.uri}", "header") try: r = await fetch_with_proxy(seg.absolute_uri, proxy_prefix) data = await r.bytes() ts_info = await window.parseSegmentTimestamps(Uint8Array.new(data).buffer) # ts_info: { pts: {first, last}, dts: {first, last} } d_start = ts_info.dts.first d_end = ts_info.dts.last p_start = ts_info.pts.first # DTS를 우선으로 하되, DTS가 없으면 PTS 사용 main_start = d_start if d_start is not None else p_start if main_start is None: log(" [!] DTS/PTS 데이터 없음", "warning") continue log(f" DTS: {d_start if d_start is not None else 'N/A'} | PTS: {p_start if p_start is not None else 'N/A'}", "info") if last_dts_end is not None: diff = main_start - last_dts_end if abs(diff) > 0.1: log(f" 🚨 {'Gap' if diff > 0 else 'Overlap'}: {diff:.3f}s", "error") else: log(f" ✅ 연속성 정상 ({diff:.3f}s)", "success") # 다음 분석을 위해 현재 세그먼트의 끝 시간 저장 (DTS 우선) last_dts_end = d_end if d_end is not None else ts_info.pts.last window.updateProgress(int((i+1)/count * 100)) except Exception as e: log(f" [!] 세그먼트 오류: {e}", "error") log("[*] 모든 분석 완료", "header")
